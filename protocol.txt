// 预定义类型
position:{
	x:number,
	y:number
}


// 请求生成地图
req:'map.createMap'
params:{
	// 地图大小
	size:number
}
res:{
	// 请求是否成功
	flag:boolean,
	data:{
		// 地图数据
		map:{
			posi:position,
			count:number
		}[]
	}
}


// 拖拽
req:'node.drag'
params:{
	from:position,
	to:position
}
res:{
	flag:boolean,
	data:{
		// 是否能移动
		isSucc:boolean,
		// 前进还是回退
		action:'add'|'sub',
		// 路线
		path:position[],
		// 是否用尽count
		isFull:boolean
	}
}

// 判断是否胜利
req:'game.isWin'
params:null
res:{
	flag:boolean,
	data:{
		isWin:boolean
	}
}
